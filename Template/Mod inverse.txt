long long factorial(int n)
{
    long long result = 1;
    for (ll i = 2; i <= n; ++i) {
        result = (result * i) % mod;
    }
    return result;
}


long long mod_exp(long long base, long long exp)
{
    long long result = 1;
    while (exp > 0) {
        if (exp % 2 == 1) {
            result = (result * base) % mod;
        }
        base = (base * base) % mod;
        exp /= 2;
    }
    return result;
}


long long mod_inverse(long long n)
{
    return mod_exp(n, mod - 2);  
}
